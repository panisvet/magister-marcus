\begin{lessonbox}
\textbf{Adding means joining groups together.}\\[4pt]
\begin{itemize}[leftmargin=1.5em, itemsep=3pt]
  \item \textbf{Adding 0} means \emph{nothing joins} — the number stays the same.\\
        \hspace{1em} $4 + 0 = 4$ \quad\quad \countdots{4} $+$ \textit{(nothing)} $= 4$
  \item \textbf{Adding 1} means \emph{one more} — count up one.\\
        \hspace{1em} $4 + 1 = 5$ \quad\quad \countdots{4} $+$ \countdots[scholagreen]{1} $= 5$
\end{itemize}
\end{lessonbox}
