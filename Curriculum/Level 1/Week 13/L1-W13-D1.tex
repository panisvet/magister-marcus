\begin{instructbox}
\textbf{Count Back} to subtract — start at the big number, hop \emph{left} that many steps.
\begin{itemize}
  \item \textbf{$-4$:} \quad $7-4=?$ \quad start at 7: \textit{6, 5, 4, \textbf{3}} \quad Answer: 3
  \item \textbf{$-5$:} \quad $8-5=?$ \quad start at 8: \textit{7, 6, 5, 4, \textbf{3}} \quad Answer: 3
\end{itemize}
\textit{Tip: Hold up your fingers and put one down for each hop.}
\end{instructbox}

\textbf{\color{sd-green}Addition Helps Subtraction!}\\[3pt]
