\begin{lessonbox}
\textbf{What is Subtraction?}\\[4pt]
Subtraction means \textbf{taking away} from a group to find how many \textit{are left}.\\[6pt]
We use the sign \textbf{−} to mean ``minus'' or ``take away.''\\[8pt]
\begin{center}
\begin{tikzpicture}[scale=0.9]
  % start: 6 dots
  \foreach \i in {1,...,6}{\filldraw[scholablue](\i*0.5,0.1)circle(0.18cm);}
  % cross out first 2
  \foreach \i in {1,2}{
    \draw[very thick, scholared](\i*0.5-0.18,-0.1) -- (\i*0.5+0.18,0.32);
    \draw[very thick, scholared](\i*0.5-0.18,0.32) -- (\i*0.5+0.18,-0.1);
  }
  \node at (3.9,0.1){\Huge\textcolor{scholared}{$-$}};
  \node[above=2pt] at (4.8,0.18){\Large 2};
  \node at (5.7,0.1){\Huge\textcolor{scholared}{$=$}};
  \foreach \i in {1,...,4}{\filldraw[scholablue](\i*0.5+6.1,0.1)circle(0.18cm);}
\end{tikzpicture}
\end{center}
\begin{center}
  \Large $6 \;-\; 2 \;=\; 4$
\end{center}
\medskip
\noindent We say: \textit{``Six minus two equals four.''}
\end{lessonbox}
