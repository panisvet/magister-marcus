\begin{lessonbox}
\textbf{How to solve a story problem:}
\begin{enumerate}[leftmargin=2em, itemsep=2pt]
  \item Read carefully. What is being counted?
  \item Draw the groups if it helps.
  \item Use the number line or counting to find your answer.
  \item Write the number on the answer line.
\end{enumerate}
\end{lessonbox}

\bigskip

\noindent\textcolor{scholared}{\large\textbf{Story Problems}} \hfill \textit{(Problems 1–10)}

\bigskip

% P1
\begin{storybox}
\textbf{1.} Peter the rabbit has \textbf{7} carrots in his basket. His sister has \textbf{9} carrots.\\
Who has \textbf{more} carrots? \hfill \answerline\\[4pt]
Write a comparison: $\quad 7 \quad$ \symbox $\quad 9$
\end{storybox}

\bigskip

% P2
\begin{storybox}
\textbf{2.} There are \textbf{6} blue birds on the fence and \textbf{6} red birds on the fence.\\
Are there \textbf{more} blue birds, \textbf{more} red birds, or the \textbf{same}? \hfill \answerline
\end{storybox}

\bigskip

% P3
\begin{storybox}
\textbf{3.} A baker bakes \textbf{8} rolls in the morning and \textbf{10} rolls in the afternoon.\\
Which baking was \textbf{bigger}? \hfill \answerline\\[4pt]
How many \textbf{more} rolls did the baker make in the afternoon? \hfill \answerline
\end{storybox}

\bigskip

% P4
\begin{storybox}
\textbf{4.} There are \textbf{10} fish in a pond. The heron eats some. Now there are \textbf{7}.\\
How many fish did the heron eat? \hfill \answerline
\end{storybox}

\bigskip

% P5
\begin{storybox}
\textbf{5.} Anna picks \textbf{9} flowers. She gives \textbf{2} to her mother.\\
Draw the 9 flowers, then cross out 2. How many are left? \hfill \answerline\\[18pt]

\noindent\textbf{7.} John has \textbf{8} marbles. He finds \textbf{0} more. How many does he have now? \hfill \answerline

\noindent\textbf{8.} Which is greater — the number of days in a week (7) or 6? \hfill \answerline
