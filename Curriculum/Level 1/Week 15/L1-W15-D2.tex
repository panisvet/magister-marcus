\begin{instructbox}
\textbf{How to solve a word problem:}
\begin{enumerate}[leftmargin=2em,itemsep=2pt]
  \item Read carefully — what is being \emph{joined} or \emph{taken away}?
  \item Write the number sentence.
  \item Solve and write the answer with its label (apples, children, etc.).
\end{enumerate}
\end{instructbox}

\bigskip
\noindent{\color{sd-blue}\large\textbf{Word Problems}} \hfill \textit{(Problems 1–10)}

\bigskip
\prob There are 6 kittens in a basket. 3 more climb in. How many kittens are there now?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob A baker had 10 loaves. She sold 7. How many loaves remain?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob Peter caught 4 butterflies. He let 4 go free. How many does he have?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob There are 5 red flowers and 4 yellow flowers in a vase. How many flowers in all?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob Mary had 9 ribbons. She gave some to her friend. Now she has 2. How many did she give away?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob Father planted 7 seeds. Rain washed away 3. How many seeds are left?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob Eight children sat at the table. Then 2 more joined them. How many children in all?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob A pond had 10 frogs. Some jumped away. Now there are 6. How many jumped away?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob Grandmother baked 3 apple pies and 6 cherry pies. How many pies in all?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank

\medskip
\prob A shelf had 9 books. A child took some home. Now there are 4. How many were taken?\\[6pt]
\hspace{2em}Number sentence:\ansblanklg \quad Answer:\ansblank
