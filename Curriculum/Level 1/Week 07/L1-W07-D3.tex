\begin{lessonbox}
\textbf{Part + Part = Whole}\\[4pt]
Every addition has two \textbf{parts} that join to make a \textbf{whole}.

\begin{center}
\begin{tikzpicture}[scale=0.9]
  % part 1 box
  \draw[thick, scholablue] (0,0) rectangle (2.5,1.2);
  \node at (1.25,0.6) {\large Part 1};
  % plus
  \node at (3,0.6) {\LARGE\textcolor{scholared}{$+$}};
  % part 2 box
  \draw[thick, scholablue] (3.5,0) rectangle (6,1.2);
  \node at (4.75,0.6) {\large Part 2};
  % equals
  \node at (6.5,0.6) {\LARGE\textcolor{scholared}{$=$}};
  % whole box
  \draw[thick, scholared] (7,0) rectangle (10,1.2);
  \node at (8.5,0.6) {\large\textbf{Whole}};
\end{tikzpicture}
\end{center}
\end{lessonbox}
