\lessonheader{3}{13}{Before, After, and Between — Numbers 0–5}

\begin{instructbox}
\noindent\textbf{\color{sd-blue}Instruction} \textit{(read aloud):}\\[4pt]
On the number line, every number has neighbors.
The number \textbf{before} is one less.
The number \textbf{after} is one more.
The number \textbf{between} two numbers sits in the middle.
\end{instructbox}

\vspace{6pt}

\noindent\textbf{\color{sd-blue}Practice}
\vspace{6pt}

\prob What number comes \textbf{after} 2? \quad\ansblank

\vspace{8pt}
\prob What number comes \textbf{before} 4? \quad\ansblank

\vspace{8pt}
\prob What number is \textbf{between} 1 and 3? \quad\ansblank

\vspace{8pt}
\prob What number comes \textbf{after} 0? \quad\ansblank

\vspace{8pt}
\prob What number comes \textbf{before} 1? \quad\ansblank

\vspace{8pt}
\prob What number is \textbf{between} 3 and 5? \quad\ansblank

\vspace{8pt}
\prob Fill in the blanks:\\[6pt]
\hspace{1cm}\ansblank \quad 2 \quad \ansblank\\[8pt]
\hspace{1cm}\ansblank \quad 4 \quad \ansblank\\[8pt]
\hspace{1cm}\ansblank \quad 0 \quad \ansblank

\vspace{8pt}
\prob \textit{(Teacher reads)} Three rabbits were hopping in a row.
The middle rabbit is between the first and the last.
If the first is rabbit number 1 and the last is rabbit number 3,
what number is the middle rabbit? \quad\ansblank

\vspace{8pt}
\prob Circle the number that is \textbf{between} 2 and 4:\\[6pt]
\hspace{2cm}{\Large 1 \quad 3 \quad 5}

\vspace{8pt}
\prob Write all the numbers from 0 to 5 in order:\\[6pt]
\hspace{1cm}\ansblank\quad\ansblank\quad\ansblank\quad\ansblank\quad\ansblank\quad\ansblank
